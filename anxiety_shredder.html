/**

* 动态打字机效果，逐字显示文本，并跳过 HTML 标签（如 <b>）。

* @param {string} text - 包含 HTML 标签的文本字符串。

* @param {HTMLElement} element - 目标 DOM 元素。

* @param {number} delay - 字符间的延迟（毫秒）。

* @returns {Promise<void>}

*/

function typeWriter(text, element, delay = 35) {

element.innerHTML = '';

let i = 0;

const type = async () => {

while (i < text.length) {

let char = text.charAt(i);

if (char === '<') {

// 遇到 < 符号，视为标签开始，找到标签结束 >

let endTagIndex = text.indexOf('>', i);

if (endTagIndex !== -1) {

// 立即将整个标签（如 <b> 或 </b>）添加到 DOM 中

const tag = text.substring(i, endTagIndex + 1);

element.innerHTML += tag;

i = endTagIndex + 1;

continue; // 跳过标签，避免延迟

}

}

// 对于普通字符，添加字符并等待延迟

element.innerHTML += char;

i++;

// 使用 Promise/setTimeout 实现延迟

await new Promise(r => setTimeout(r, delay));

}

};

 

return type();

}

}

}
